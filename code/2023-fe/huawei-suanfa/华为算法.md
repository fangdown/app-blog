    const diff = target - numbers[i];
    if (arr[diff] == undefined) {

    if (/[A-Z]/.test(str.charAt(i))) t3++;
    if (/[^a-z^A-Z^0-9]/.test(str.charAt(i))) t4++;
  }

    const x = str.slice(i, i + 3);

      res.push(temp);
      temp = cur;

function maxDepth(s) {
  const stack = [];
  let maxDepth = 0;
  for (let i = 0; i < s.length; i++) {
    if (s[i] === "(") {
      stack.push(s[i]);
    } else if (s[i] === ")") {
      stack.pop();

}

function permutation() {
  let res = [];
  let path = [];
  let used = new Array(s.length).fill(false);
  const dfs = (s, path, used) => {
    if (path.length === s.length) {
      res.push(path.join(""));
      return;
    }
    for (let i = 0; i < s.length; i++) {
      if (used[i]) continue;
      if (i > 0 && s[i] === s[i - 1] && !used[i - 1]) continue;
      path.push(s[i]);
      used[i] = true;
      dfs(s, path, used);
      path.pop();
      used[i] = false;
    }
  };
  dfs(s, path, used);
  return res;
}
