# Rollup

是一款ES Modules打包器，和webpack类型，但是小巧，适合打包库

// user.js
export const name = 'fangdown'

// index.js
import {name} from './user'
export const showName = (name) =>{
  console.log(name)
}
showName(name)

执行打包

 npx rollup code/fe-engin/Rollup/src/index.js --file code/fe-engin/Rollup/dist/bundle.js
  • 生成bundle.js
const name = 'fangdown';

const showName = (name) =>{
  console.log(name);
};
showName(name);

export { showName };
